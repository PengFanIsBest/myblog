# 小程序双击事件

# context是点击的目标对象 option配置单机和双击的事件执行操作 eve(参数)传递事件源对象

# 小程序点击事件执行调用event事件传入参数 通过判断点击第一次和第二次的时间，来判断执行时双击或者单机事件

# 原理分析：就是通过setTimeout()来执行单机，双击事件时就取消定时器，此时调用双击事件的回调就行了

let store = {}
let date = new Date();
let time, item;
let event = function (context, option, eve) {
  //双击事件间隔 time
  let opt = {
    time: 300,
    click: function (eve) {
      console.log("单击")
    },
    dblclick: function (eve) {
      console.log("双击")
    }
  }
  if (option) {
    for (item in option) {
      opt[item] = option[item]
    }
  }
  time = opt.time;
  if (!store[0]) {
    store[0] = date.getTime();
    // 定时器   执行单击事件
    TimeOut = setTimeout(() => {

      opt.click.call(context)
      console.log("执行单击事件倒计时300ms")
      //重置判断条件
      store[0] = null;
    }, time)
  } else {
    let resout = date.getTime() - store[0]
    if (resout < time) {
      //取消单击事件执行双击事件   重置判断条件
      opt.dblclick.call(context, eve)
      // console.log("执行双击事件")
      clearTimeout(TimeOut)
      store[0] = null;
    }
  }
}
